\documentclass[border=10pt]{standalone}

\usepackage{tikz}
\usetikzlibrary{shapes,arrows}

\begin{document}

\tikzstyle{block} = [rectangle, draw, node distance=3cm,
    text width=6em, text centered, rounded corners, minimum height=4em]
\tikzstyle{line} = [draw, -latex']
    
\begin{tikzpicture}
 
    \coordinate (TL) at (-4,  3);
    \coordinate (TR) at ( 4,  3);
    \coordinate (BL) at (-4, -3);
    \coordinate (BR) at ( 4, -3);
    \coordinate (o) at ( 0,  0);
    \coordinate (x) at ( 3,  0);
    \coordinate (y) at ( 0,  2);
    
    \draw (BL) rectangle (TR);
    
    \draw [->] (o) node [below left] {$(0, 0)$} -- (x) node [right] {$x$};
    \draw [->] (o) -- (y) node [above] {$y$};
    
    \node [above left] at (TL) {$(-1, 1)$};
    \node [below left] at (BL) {$(-1, -1)$};
    \node [above right] at (TR) {$(1, 1)$};
    \node [below right] at (BR) {$(1, -1)$};
 
\end{tikzpicture}
\end{document}
